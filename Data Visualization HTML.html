df8<-read.csv("df8.csv")
#bar plot for age vs outcome
age <- table(df8$Outcome,df8$Age.Group)
barplot(age, main="Age and COVID Cases",
        xlab="Age Group", col=c("lightblue","pink"),
        legend = rownames(age), beside=TRUE)

#bar plot for outbreak vs outcome
out <- table(df8$Outcome,df8$Outbreak.Associated)
barplot(out, main="Outbreak Type and COVID Cases",
        xlab="Outbreak Type", ylab="Frequency", col=c("lightblue","pink"),
        legend = rownames(out), beside=TRUE)

#bar plot for source vs outcome
sou <- table(df8$Outcome,df8$Source.of.Infection)
barplot(sou, main="Source of Infection and COVID Cases",
        xlab="Source of Infection", col=c("lightblue","pink"),
        legend = rownames(sou), beside=TRUE)
